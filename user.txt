table: User
    fields:
        uuid
        profile_uid(o2o)
        email
        pass
        phone_number
        status
        type
        username
        firstname
        lastname
        avatar

    type:
        super admin
        intitute admin
        campus admin
        student
        teacher
        basic user(default)
        parent
        staff
    
    status:
        NOT_VERIFIED    
        ACTIVE
        DEACTIVED
        PASS_RESET_REQUIRED
        BANNED
        INACTIVE

        note: valid status transition: default status NOT_VERIFIED until a user verifies the email/phone_number
                                verified users are ACTIVE, from ACTIVE one can not go to unverified, but others can
            valid type transition: default type is basic_user, from that one can be any except super user
